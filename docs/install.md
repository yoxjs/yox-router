## CDN

```html
<script src="https://unpkg.com/yox"></script>
<script src="https://unpkg.com/yox-router"></script>
<script>
  Yox.use(YoxRouter)
</script>
```

## NPM

```shell
npm install yox
npm install yox-router
```

```js
import Yox from 'yox'
import YoxRouter from 'yox-router'
Yox.use(YoxRouter)
```

## YARN

```shell
yarn add yox
yarn add yox-router
```

```js
import Yox from 'yox'
import YoxRouter from 'yox-router'
Yox.use(YoxRouter)
```
